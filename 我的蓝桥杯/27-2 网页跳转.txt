//#include "header.h"	//AnycodeX includes the header.h by default, needn't cancle the #include <iostream>#include <iostream>#include <stack>#include <string>using namespace std;int main(){	int n;	cin>> n;	stack<string> backnow;	stack<string> forward;  	while(n--){	    string way,web;	    cin >> way ;	    if (way == "VISIT"){		cin>> web;	        while (!forward.empty()){ forward.pop(); }	        backnow.push(web);	        cout << backnow.top()<<endl;	    }	    else if (way=="BACK"){	        if (backnow.empty())	            cout << "Ignore"<<endl;	       else{	           forward.push(backnow.top());	           backnow.pop();	           if (backnow.empty()){	               cout << "Ignore"<<endl;	               backnow.push(forward.top());	               forward.pop();	           }else  cout << backnow.top()<<endl;	       }	    }else{	        if (forward.empty())	            cout << "Ignore"<<endl;	        else{	            if (!forward.empty()){	                cout << forward.top()<<endl;	                backnow.push(forward.top());	                forward.pop();	            }else cout << "Ignore"<<endl;	        }	    }	}		return 0;}
